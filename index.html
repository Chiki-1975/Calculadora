import React, { useState, useEffect } from 'react';

import { Calculator, DollarSign, Package, Truck, Users, TrendingUp } from 'lucide-react';

const CostCalculator = () => {

  const [costs, setCosts] = useState({

    // Costos de Materiales

    materialPrincipal: 0,

    materialesSecundarios: 0,

    herrajes: 0,

    embalaje: 0,

   

    // Costos de Producción

    manoObra: 0,

    energiaServicios: 0,

    depreciacionMaquinaria: 0,

    controlCalidad: 0,

   

    // Costos Logísticos

    transporte: 0,

    almacenamiento: 0,

    seguros: 0,

   

    // Costos Administrativos

    gastosGenerales: 0,

    marketing: 0,

    ventasComisiones: 0,

   

    // Márgenes y Otros

    margenUtilidad: 25,

    impuestos: 19,

    descuentosComerciales: 5

  });

  const [totals, setTotals] = useState({

    costoMateriales: 0,

    costoProduccion: 0,

    costoLogistico: 0,

    costoAdministrativo: 0,

    costoTotal: 0,

    precioConMargen: 0,

    impuestosAplicados: 0,

    precioFinalPublico: 0,

    precioConDescuento: 0

  });

  useEffect(() => {

    const costoMateriales = costs.materialPrincipal + costs.materialesSecundarios + costs.herrajes + costs.embalaje;

    const costoProduccion = costs.manoObra + costs.energiaServicios + costs.depreciacionMaquinaria + costs.controlCalidad;

    const costoLogistico = costs.transporte + costs.almacenamiento + costs.seguros;

    const costoAdministrativo = costs.gastosGenerales + costs.marketing + costs.ventasComisiones;

   

    const costoTotal = costoMateriales + costoProduccion + costoLogistico + costoAdministrativo;

    const precioConMargen = costoTotal * (1 + costs.margenUtilidad / 100);

    const impuestosAplicados = precioConMargen * (costs.impuestos / 100);

    const precioFinalPublico = precioConMargen + impuestosAplicados;

    const precioConDescuento = precioFinalPublico * (1 - costs.descuentosComerciales / 100);

    setTotals({

      costoMateriales,

      costoProduccion,

      costoLogistico,

      costoAdministrativo,

      costoTotal,

      precioConMargen,

      impuestosAplicados,

      precioFinalPublico,

      precioConDescuento

    });

  }, [costs]);

  const handleInputChange = (field, value) => {

    setCosts(prev => ({

      ...prev,

      [field]: parseFloat(value) || 0

    }));

  };

  const formatCurrency = (amount) => {

    return new Intl.NumberFormat('es-CO', {

      style: 'currency',

      currency: 'COP',

      minimumFractionDigits: 0,

      maximumFractionDigits: 0

    }).format(amount);

  };

  return (

    <div className="max-w-6xl mx-auto p-6 bg-gray-50 min-h-screen">

      <div className="bg-white rounded-lg shadow-lg p-6 mb-6">

        <div className="flex items-center gap-3 mb-6">

          <Calculator className="w-8 h-8 text-blue-600" />

          <h1 className="text-3xl font-bold text-gray-800">Calculadora de Costos y Precio de Venta</h1>

        </div>

       

        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">

          {/* Columna Izquierda - Inputs */}

          <div className="space-y-6">

           

            {/* Costos de Materiales */}

            <div className="bg-blue-50 p-4 rounded-lg">

              <div className="flex items-center gap-2 mb-4">

                <Package className="w-5 h-5 text-blue-600" />

                <h3 className="text-lg font-semibold text-blue-800">Costos de Materiales</h3>

              </div>

              <div className="grid grid-cols-2 gap-4">

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Material Principal</label>

                  <input

                    type="number"

                    value={costs.materialPrincipal}

                    onChange={(e) => handleInputChange('materialPrincipal', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-blue-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Materiales Secundarios</label>

                  <input

                    type="number"

                    value={costs.materialesSecundarios}

                    onChange={(e) => handleInputChange('materialesSecundarios', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-blue-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Herrajes y Accesorios</label>

                  <input

                    type="number"

                    value={costs.herrajes}

                    onChange={(e) => handleInputChange('herrajes', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-blue-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Embalaje</label>

                  <input

                    type="number"

                    value={costs.embalaje}

                    onChange={(e) => handleInputChange('embalaje', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-blue-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

              </div>

            </div>

            {/* Costos de Producción */}

            <div className="bg-green-50 p-4 rounded-lg">

              <div className="flex items-center gap-2 mb-4">

                <Users className="w-5 h-5 text-green-600" />

                <h3 className="text-lg font-semibold text-green-800">Costos de Producción</h3>

              </div>

              <div className="grid grid-cols-2 gap-4">

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Mano de Obra</label>

                  <input

                    type="number"

                    value={costs.manoObra}

                    onChange={(e) => handleInputChange('manoObra', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-green-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Energía y Servicios</label>

                  <input

                    type="number"

                    value={costs.energiaServicios}

                    onChange={(e) => handleInputChange('energiaServicios', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-green-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Depreciación Maquinaria</label>

                  <input

                    type="number"

                    value={costs.depreciacionMaquinaria}

                    onChange={(e) => handleInputChange('depreciacionMaquinaria', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-green-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Control de Calidad</label>

                  <input

                    type="number"

                    value={costs.controlCalidad}

                    onChange={(e) => handleInputChange('controlCalidad', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-green-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

              </div>

            </div>

            {/* Costos Logísticos */}

            <div className="bg-orange-50 p-4 rounded-lg">

              <div className="flex items-center gap-2 mb-4">

                <Truck className="w-5 h-5 text-orange-600" />

                <h3 className="text-lg font-semibold text-orange-800">Costos Logísticos</h3>

              </div>

              <div className="grid grid-cols-2 gap-4">

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Transporte</label>

                  <input

                    type="number"

                    value={costs.transporte}

                    onChange={(e) => handleInputChange('transporte', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-orange-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Almacenamiento</label>

                  <input

                    type="number"

                    value={costs.almacenamiento}

                    onChange={(e) => handleInputChange('almacenamiento', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-orange-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Seguros</label>

                  <input

                    type="number"

                    value={costs.seguros}

                    onChange={(e) => handleInputChange('seguros', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-orange-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

              </div>

            </div>

            {/* Costos Administrativos */}

            <div className="bg-purple-50 p-4 rounded-lg">

              <div className="flex items-center gap-2 mb-4">

                <TrendingUp className="w-5 h-5 text-purple-600" />

                <h3 className="text-lg font-semibold text-purple-800">Costos Administrativos</h3>

              </div>

              <div className="grid grid-cols-2 gap-4">

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Gastos Generales</label>

                  <input

                    type="number"

                    value={costs.gastosGenerales}

                    onChange={(e) => handleInputChange('gastosGenerales', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-purple-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Marketing y Publicidad</label>

                  <input

                    type="number"

                    value={costs.marketing}

                    onChange={(e) => handleInputChange('marketing', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-purple-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Ventas y Comisiones</label>

                  <input

                    type="number"

                    value={costs.ventasComisiones}

                    onChange={(e) => handleInputChange('ventasComisiones', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-purple-500 focus:border-transparent"

                    placeholder="0"

                  />

                </div>

              </div>

            </div>

            {/* Márgenes y Porcentajes */}

            <div className="bg-yellow-50 p-4 rounded-lg">

              <div className="flex items-center gap-2 mb-4">

                <DollarSign className="w-5 h-5 text-yellow-600" />

                <h3 className="text-lg font-semibold text-yellow-800">Márgenes y Porcentajes</h3>

              </div>

              <div className="grid grid-cols-3 gap-4">

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Margen de Utilidad (%)</label>

                  <input

                    type="number"

                    value={costs.margenUtilidad}

                    onChange={(e) => handleInputChange('margenUtilidad', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-yellow-500 focus:border-transparent"

                    placeholder="25"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Impuestos (%)</label>

                  <input

                    type="number"

                    value={costs.impuestos}

                    onChange={(e) => handleInputChange('impuestos', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-yellow-500 focus:border-transparent"

                    placeholder="19"

                  />

                </div>

                <div>

                  <label className="block text-sm font-medium text-gray-700 mb-1">Descuento Comercial (%)</label>

                  <input

                    type="number"

                    value={costs.descuentosComerciales}

                    onChange={(e) => handleInputChange('descuentosComerciales', e.target.value)}

                    className="w-full p-2 border border-gray-300 rounded-md focus:ring-2 focus:ring-yellow-500 focus:border-transparent"

                    placeholder="5"

                  />

                </div>

              </div>

            </div>

          </div>

          {/* Columna Derecha - Resultados */}

          <div className="space-y-6">

            <div className="bg-white border-2 border-gray-200 rounded-lg p-6 sticky top-6">

              <h3 className="text-xl font-bold text-gray-800 mb-6 text-center">Resumen de Costos</h3>

             

              {/* Desglose de Costos */}

              <div className="space-y-4 mb-6">

                <div className="flex justify-between items-center p-3 bg-blue-50 rounded">

                  <span className="font-medium text-blue-800">Costos de Materiales:</span>

                  <span className="font-bold text-blue-900">{formatCurrency(totals.costoMateriales)}</span>

                </div>

                <div className="flex justify-between items-center p-3 bg-green-50 rounded">

                  <span className="font-medium text-green-800">Costos de Producción:</span>

                  <span className="font-bold text-green-900">{formatCurrency(totals.costoProduccion)}</span>

                </div>

                <div className="flex justify-between items-center p-3 bg-orange-50 rounded">

                  <span className="font-medium text-orange-800">Costos Logísticos:</span>

                  <span className="font-bold text-orange-900">{formatCurrency(totals.costoLogistico)}</span>

                </div>

                <div className="flex justify-between items-center p-3 bg-purple-50 rounded">

                  <span className="font-medium text-purple-800">Costos Administrativos:</span>

                  <span className="font-bold text-purple-900">{formatCurrency(totals.costoAdministrativo)}</span>

                </div>

              </div>

              <hr className="my-6" />

              {/* Cálculo Final */}

              <div className="space-y-4">

                <div className="flex justify-between items-center p-3 bg-gray-100 rounded">

                  <span className="font-bold text-gray-800">COSTO TOTAL:</span>

                  <span className="font-bold text-gray-900 text-lg">{formatCurrency(totals.costoTotal)}</span>

                </div>

                <div className="flex justify-between items-center p-3 bg-yellow-50 rounded">

                  <span className="font-medium text-yellow-800">+ Margen ({costs.margenUtilidad}%):</span>

                  <span className="font-bold text-yellow-900">{formatCurrency(totals.precioConMargen)}</span>

                </div>

                <div className="flex justify-between items-center p-3 bg-red-50 rounded">

                  <span className="font-medium text-red-800">+ Impuestos ({costs.impuestos}%):</span>

                  <span className="font-bold text-red-900">{formatCurrency(totals.impuestosAplicados)}</span>

                </div>

                <div className="flex justify-between items-center p-4 bg-green-100 rounded-lg border-2 border-green-300">

                  <span className="font-bold text-green-800 text-lg">PRECIO FINAL AL PÚBLICO:</span>

                  <span className="font-bold text-green-900 text-2xl">{formatCurrency(totals.precioFinalPublico)}</span>

                </div>

                {costs.descuentosComerciales > 0 && (

                  <div className="flex justify-between items-center p-3 bg-blue-100 rounded">

                    <span className="font-medium text-blue-800">Precio con Descuento ({costs.descuentosComerciales}%):</span>

                    <span className="font-bold text-blue-900">{formatCurrency(totals.precioConDescuento)}</span>

                  </div>

                )}

              </div>

              {/* Métricas Adicionales */}

              <hr className="my-6" />

              <div className="space-y-2 text-sm text-gray-600">

                <div className="flex justify-between">

                  <span>Margen sobre costo:</span>

                  <span>{((totals.precioFinalPublico - totals.costoTotal) / totals.costoTotal * 100).toFixed(1)}%</span>

                </div>

                <div className="flex justify-between">

                  <span>Margen sobre precio:</span>

                  <span>{((totals.precioFinalPublico - totals.costoTotal) / totals.precioFinalPublico * 100).toFixed(1)}%</span>

                </div>

                <div className="flex justify-between">

                  <span>Punto de equilibrio (unidades):</span>

                  <span>{totals.costoTotal > 0 ? Math.ceil(totals.costoTotal / (totals.precioFinalPublico - totals.costoTotal)) : 0}</span>

                </div>

              </div>

            </div>

          </div>

        </div>

      </div>

    </div>

  );

};

export default CostCalculator;
